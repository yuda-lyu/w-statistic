async function arrLogNormHist(arr, opt = {}) {

    //check arr
    if (!isarr(arr)) {
        return Promise.reject('arr is not an array')
    }
    if (size(arr) === 0) {
        return Promise.reject('arr is not an effective array')
    }

    //arrNormHist for normal
    let r = await arrNormHist(arr, opt)
    // console.log('arrNormHist for normal', r)

    //取大於0數值並取log
    let rs = arrFilterByPnumAndToLog(arr)
    // console.log('arrFilterByPnumAndToLog', rs)

    //arrNormHist for log-normal
    let r_log = await arrNormHist(rs) //log不能使用opt否則會使用到外部指定最小最大值
    // console.log('arrNormHist for log-normal', r_log)

    //curves
    let curves = map(r_log.curves, (v) => {
        return {
            x: Math.exp(v.x),
            pdf: v.pdf
        }
    })

    //修正pdf
    let rr = r.ratioForCountToPdf / r_log.ratioForCountToPdf
    curves = map(curves, (v) => {
        v.pdf *= rr
        return v
    })

    //arrGeometricAverage
    let avg_geo = arrGeometricAverage(r.arr)

    //res
    let res = {
        avg: r.avg,
        std: r.std,
        avg_geo, //幾何平均值
        avg_log: r_log.avg, //轉至log座標的平均值
        std_log: r_log.std, //轉至log座標的標準差
        arr: r.arr,
        min: r.min,
        max: r.max,
        barWidth: r.barWidth,
        ratioForCountToPdf: r.ratioForCountToPdf,
        bins: r.bins,
        curves, //基於log座標反算的對數常態分佈曲線
    }

    return res
}
