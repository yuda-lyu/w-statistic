function ltdtMovingAverage(ltdt, key, opt) {

    //check
    if (!isarr(ltdt)) {
        return []
    }
    if (!isestr(key)) {
        return []
    }

    //rs
    let rs = map(ltdt, key)

    //arrMovingAverage
    let r = arrMovingAverage(rs, opt)

    //update
    ltdt = map(ltdt, (v, k) => {
        v[key] = r[k]
        return v
    })

    return ltdt
}
