  • arrCount.mjs, line 49

計算陣列內有效數字依照不重複值進行群組化後，回傳各值之出現次數值

Unit Test: Github

Example
let arr

arr = ['abc', '-2.5', -2.5, '-1', -1, '-0.1', -0.1, '0', 0, '0.1', 0.1, '1', 1, '2.5', 2.5, 22.5, 'xyz']
console.log(arrCount(arr))
// => [
//   { key: '2.5', count: 2 },
//   { key: '0.1', count: 2 },
//   { key: '-0.1', count: 2 },
//   { key: '-1', count: 2 },
//   { key: '-2.5', count: 2 },
//   { key: '1', count: 2 },
//   { key: '0', count: 2 },
//   { key: '22.5', count: 1 }
// ]

arr = ['abc', '0', 0, '0.1', 0.1, '1', 1, '2.5', 2.5, 22.5, 'xyz']
console.log(arrCount(arr))
// => [
//   { key: '2.5', count: 2 },
//   { key: '0.1', count: 2 },
//   { key: '1', count: 2 },
//   { key: '0', count: 2 },
//   { key: '22.5', count: 1 }
// ]
